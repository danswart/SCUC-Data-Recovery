SCUC Dataset Match Audit

Find every plausible SCUC Snapshot dataset and compare it with the canonical ladder

What this workbook does

This is the read-only half of the recovery workflow. It searches the entire R working directory, identifies plausible SCUC Snapshot datasets by filename and structure, reads every supported table or workbook sheet, and compares normalized table content with every file in:

  • 01 reference canonical
  • 02 reference pre-canonical

It writes an auditable manifest to 04 project inventory outputs. It never copies, moves, renames, overwrites, or deletes source data. Run 01_scuc_stage_classified_copies.qmd only after reviewing the manifest.

The word exact below means exact normalized table content: same columns, rows, and cell values after harmless representation differences are removed (column/row order, surrounding whitespace, common missing-value spellings, and numeric formatting such as 2 versus 2.0). A byte-for-byte SHA-256 match is also reported separately.

2. Comparison helpers

Each Excel sheet is treated as a separate table. A workbook is assigned its best matching sheet, while the sheet-level results remain in the detailed report.

Code 
clean_name <- function(x) {
  x |>
    enc2utf8() |>
    trimws() |>
    tolower() |>
    gsub("^\\ufeff", "", x = _) |>
    gsub("[^a-z0-9]+", "_", x = _) |>
    gsub("^_|_$", "", x = _)
}

normalize_cell <- function(x) {
  y <- trimws(enc2utf8(as.character(x)))
  missing <- is.na(y) | tolower(y) %in% c("", "na", "n/a", "null", "nan", "#n/a")
  numeric_like <- !missing & grepl("^[+-]?(?:[0-9]+(?:\\.[0-9]*)?|\\.[0-9]+)(?:[eE][+-]?[0-9]+)?$", y, perl = TRUE)
  if (any(numeric_like)) {
    z <- suppressWarnings(as.numeric(y[numeric_like]))
    y[numeric_like] <- format(z, scientific = FALSE, trim = TRUE, digits = 15)
  }
  y[missing] <- "<NA>"
  y
}

normalize_table <- function(x) {
  x <- as.data.frame(x, stringsAsFactors = FALSE, check.names = FALSE)
  names(x) <- make.unique(clean_name(names(x)), sep = "__duplicate_")
  x[] <- lapply(x, normalize_cell)
  x <- x[, order(names(x)), drop = FALSE]
  if (nrow(x)) {
    row_text <- do.call(paste, c(unname(x), list(sep = "\u001f")))
    x <- x[order(row_text, method = "radix"), , drop = FALSE]
  }
  rownames(x) <- NULL
  x
}

table_fingerprint <- function(x) {
  digest::digest(serialize(normalize_table(x), NULL, version = 2), algo = "sha256")
}

read_tables <- function(path) {
  ext <- tolower(fs::path_ext(path))
  tryCatch({
    if (ext %in% c("csv", "tsv")) {
      delim <- if (ext == "tsv") "\t" else ","
      list(data = readr::read_delim(path, delim = delim, col_types = readr::cols(.default = "c"),
                                    name_repair = "minimal", progress = FALSE, show_col_types = FALSE,
                                    trim_ws = FALSE))
    } else if (ext == "xls") {
      stop("Legacy .xls retained for manual review; not parsed automatically")
    } else if (ext %in% c("xlsx", "xlsm")) {
      sheets <- readxl::excel_sheets(path)
      stats::setNames(lapply(sheets, function(s) {
        readxl::read_excel(path, sheet = s, col_types = "text", .name_repair = "minimal")
      }), sheets)
    } else if (ext == "rds") {
      object <- readRDS(path)
      if (!is.data.frame(object)) stop("RDS object is not a data frame")
      list(data = object)
    } else stop("Unsupported extension")
  }, error = function(e) structure(list(), read_error = conditionMessage(e)))
}

profile_table <- function(x) {
  nm <- clean_name(names(x))
  has_year_columns <- sum(grepl("^(19|20)[0-9]{2}$", nm)) >= 3L
  has_long_pair <- any(nm %in% c("date", "year")) && "value" %in% nm
  identity_columns <- c("grouping", "section", "sort", "units", "grade", "level_achieved")
  tibble::tibble(
    rows = nrow(x), columns = ncol(x),
    normalized_columns = paste(sort(nm), collapse = ";"),
    identity_column_count = sum(identity_columns %in% nm),
    has_year_columns = has_year_columns,
    has_long_pair = has_long_pair,
    structural_snapshot_signal = sum(identity_columns %in% nm) >= 2L && (has_year_columns || has_long_pair)
  )
}

4. Enumerate and structurally screen candidates

The audit does not assume that filenames are reliable. A table is included when either its path contains a strong Snapshot phrase or its columns resemble the canonical Snapshot schema. Generic SCUC datasets about bonds, surveys, or Census data are therefore reported as screened-out instead of being mixed into the canonical comparison.

5. Classify exact matches and earlier variants

Priority is canonical match, then pre-canonical match, then an earlier/unmatched Snapshot variant. Matching is based on the best sheet in a workbook. Files already inside this recovery project are inventoried but are not staged again.

Code 
sheet_matches <- candidate_tables |>
  dplyr::filter(in_scope) |>
  dplyr::left_join(
    reference_tables |>
      dplyr::select(reference_file, reference_sheet, reference_class,
                    reference_byte_sha256 = byte_sha256,
                    reference_content_sha256 = content_sha256,
                    reference_rows = rows, reference_columns = columns,
                    reference_normalized_columns = normalized_columns),
    by = c("content_sha256" = "reference_content_sha256"),
    relationship = "many-to-many"
  ) |>
  dplyr::mutate(
    byte_exact = !is.na(reference_file) & byte_sha256 == reference_byte_sha256,
    content_exact = !is.na(reference_file),
    match_rank = dplyr::case_when(
      reference_class == "canonical" & byte_exact ~ 1L,
      reference_class == "canonical" & content_exact ~ 2L,
      reference_class == "pre-canonical" & byte_exact ~ 3L,
      reference_class == "pre-canonical" & content_exact ~ 4L,
      TRUE ~ 5L
    )
  )

best_sheet <- sheet_matches |>
  dplyr::group_by(source_path) |>
  dplyr::arrange(match_rank, dplyr::desc(rows), source_sheet, .by_group = TRUE) |>
  dplyr::slice(1L) |>
  dplyr::ungroup() |>
  dplyr::mutate(
    classification = dplyr::case_when(
      reference_class == "canonical" ~ "06 canonical data set matches",
